```prolog
covered(C) :-
    % Check service type
    claim_service_type(C, ivf_for_fertility_preservation),
    
    % Check location of service received
    claim_location_service_received(C, Location),
    member(Location, [stanford_health_care, menlo_medical_clinic, sutter_health_palo_alto_medical_foundation, tier_two_in_network]),
    
    % Check fertility preservation eligibility
    claim_patient_is_believed_to_be_fertile(C, yes),
    claim_planned_service(C, PlannedService),
    member(PlannedService, [chemotherapy, pelvic_radiotherapy, gonadotoxic_therapy, orchiectomy, oophorectomy, hysterectomy, orchiectomy_for_treatment_of_disease, oophorectomy_for_treatment_of_disease, other_treatment_demonstrated_to_result_in_infertility]),
    
    % Check egg retrieval likelihood for successful pregnancy
    claim_will_be_retrieving_eggs_for_use(C, yes),
    not((most_recent_day_3_fsh_test(C, fsh_test(FSH_Level, _, _)), FSH_Level >= 19)),
    
    % Check medical necessity
    claim_inseminations_not_expected_to_be_effective_and_ivf_expected_to_be_only_effective_treatment(C, yes),
    
    % Check ovulation induction, cycles, and ART services
    claim_num_cycles_oral_or_injectable_ovulation_induction(C, NumCycles),
    patient_age_at_time_of_service(C, Age),
    (Age =< 37 -> NumCycles >= 3 ; Age >= 38),
    
    % Check age-related coverage limitations
    not((
        Age >= 40,
        (most_recent_day_3_fsh_test(C, fsh_test(FSH_Level_Age40, _, _)), FSH_Level_Age40 >= 19),
        day_3_fsh_tests_since_age_40_in_claim(C, Tests_Since40),
        exists_test_with_level_at_least_19(Tests_Since40)
    )),
    
    % Check ART limit per policy year
    claim_num_other_covered_art_services(C, NumART),
    NumART =< 1.
```
