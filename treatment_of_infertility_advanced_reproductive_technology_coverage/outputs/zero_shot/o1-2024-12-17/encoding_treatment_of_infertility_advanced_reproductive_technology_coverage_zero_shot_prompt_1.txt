covered(C) :-
    /* Must be one of the fertility‑preservation ART services */
    claim_service_type(C, ServiceType),
    ( ServiceType = ivf_for_fertility_preservation
    ; ServiceType = cryopreservation_for_fertility_preservation ),
    /* Must be in‑network (not 'other') */
    claim_location_service_received(C, Location),
    Location \= other,
    /* Patient must be believed to be fertile */
    claim_patient_is_believed_to_be_fertile(C, yes),
    /* Must be planning a service that results in infertility */
    claim_planned_service(C, PlannedService),
    member(
      PlannedService,
      [ chemotherapy,
        pelvic_radiotherapy,
        gonadotoxic_therapy,
        orchiectomy,
        oophorectomy,
        hysterectomy,
        orchiectomy_for_treatment_of_disease,
        oophorectomy_for_treatment_of_disease,
        other_treatment_demonstrated_to_result_in_infertility
      ]
    ),
    /* If eggs will be retrieved, must meet FSH/ovarian‑responsiveness criteria */
    ( claim_will_be_retrieving_eggs_for_use(C, yes) ->
        fsh_ok(C)
    ; true
    ).

/* Checks all FSH / ovarian‑responsiveness rules per the infertility policy */
fsh_ok(C) :-
    patient_age_at_time_of_service(C, Age),
    (   /* For women younger than 39: most recent day‑3 FSH < 19, no 6‑month requirement */
        Age < 39 ->
        most_recent_day_3_fsh_test(C, fsh_test(Level, _, _)),
        Level < 19

    ;   /* For women exactly 39: most recent day‑3 FSH < 19 and obtained within prior 6 months */
        Age < 40 ->
        most_recent_day_3_fsh_test(C, fsh_test(Level, TestDate, _)),
        claim_date_service_received(C, ServiceDate),
        leq_n_months_diff(TestDate, ServiceDate, 6),
        Level < 19

    ;   /* For women 40 or older */}
        most_recent_day_3_fsh_test(C, fsh_test(MostRecentLevel, TestDate, _)),
        claim_date_service_received(C, ServiceDate),
        leq_n_months_diff(TestDate, ServiceDate, 6),
        claim_patient_has_premature_ovarian_failure(C, HasPOF),
        (   HasPOF = yes
        ->  /* If premature ovarian failure, only check most recent < 19 */
            MostRecentLevel < 19
        ;   /* Otherwise, all day‑3 FSH tests since age 40 must be < 19 */
            day_3_fsh_tests_since_age_40_in_claim(C, Tests),
            \+ exists_test_with_level_at_least_19(Tests)
        )
    ).
