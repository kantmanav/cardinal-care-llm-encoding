covered(C) :-
    % 1) Must be an ART service for fertility preservation
    claim_service_type(C, ServiceType),
    ( ServiceType = ivf_for_fertility_preservation
    ; ServiceType = cryopreservation_for_fertility_preservation ),

    % 2) Must not be out of network
    claim_location_service_received(C, Location),
    Location \= other,

    % 3) Patient must be believed to be fertile
    claim_patient_is_believed_to_be_fertile(C, yes),

    % 4) Must have a planned service that results in infertility
    claim_planned_service(C, Planned),
    member(Planned, [
       chemotherapy,
       pelvic_radiotherapy,
       gonadotoxic_therapy,
       orchiectomy,
       oophorectomy,
       hysterectomy,
       orchiectomy_for_treatment_of_disease,
       oophorectomy_for_treatment_of_disease,
       other_treatment_demonstrated_to_result_in_infertility
    ]),

    % 5) Must satisfy FSH criteria if retrieving own eggs
    fsh_criteria(C).

fsh_criteria(C) :-
    claim_will_be_retrieving_eggs_for_use(C, no).

fsh_criteria(C) :-
    claim_will_be_retrieving_eggs_for_use(C, yes),
    patient_age_at_time_of_service(C, Age),
    (   Age < 39
    ->  most_recent_day_3_fsh_test(C, fsh_test(Level, _, _)),
        Level < 19
    ;   Age < 40
    ->  check_test_within_six_months(C),
        most_recent_day_3_fsh_test(C, fsh_test(Level, _, _)),
        Level < 19
    ;   Age >= 40
    ->  check_test_within_six_months(C),
        (   claim_patient_has_premature_ovarian_failure(C, yes)
        ->  most_recent_day_3_fsh_test(C, fsh_test(Level, _, _)),
            Level < 19
        ;   day_3_fsh_tests_since_age_40_in_claim(C, Tests),
            \+ exists_test_with_level_at_least_19(Tests)
        )
    ).

check_test_within_six_months(C) :-
    claim_date_service_received(C, ServiceDate),
    most_recent_day_3_fsh_test(C, fsh_test(_, TestDate, _)),
    leq_n_months_diff(TestDate, ServiceDate, 6).
