```prolog
excluded_location(StanfordHealthCare) :- fail.
excluded_location(MenloMedicalClinic) :- fail.
excluded_location(SutterHealthPaloAltoMedicalFoundation) :- fail.
excluded_location(TierTwoInNetwork) :- fail.
excluded_location(other).

valid_trials_performed(C) :-
    claim.age(C, Age),
    claim.trial_of_egg_sperm_contact_months(C, Months),
    (
        (Age < 35, Months >= 12);
        (Age >= 35, Age < 40, Months >= 6);
        (Age >= 40, Months >= 3)
    ).

ovarian_reserve_met(C) :-
    claim.age(C, Age),
    claim.day3_fsh(C, FSH),
    (
        (Age < 40, FSH < 19);
        (Age >= 39, FSH < 19)
    ).

under_art_trial_threshold(C) :-
    claim.num_other_covered_art_services(C, Num),
    Num < 1.

eligible_art_service(C) :-
    claim.patient_is_believed_to_be_fertile(C, yes),
    claim.will_be_retrieving_eggs_for_use(C, yes),
    ovarian_reserve_met(C),
    under_art_trial_threshold(C).

valid_antigenic_treatment(C) :-
    claim.planned_service(C, chemotherapy);
    claim.planned_service(C, pelvic_radiotherapy);
    claim.planned_service(C, gonadotoxic_therapy);
    claim.planned_service(C, orchiectomy_for_treatment_of_disease);
    claim.planned_service(C, oophorectomy_for_treatment_of_disease);
    claim.planned_service(C, other_treatment_demonstrated_to_result_in_infertility).

eligible_for_fertility_preservation(C) :-
    claim.patient_is_believed_to_be_fertile(C, yes),
    valid_antigenic_treatment(C),
    eligible_art_service(C).

covered_ivf_for_fertility_preservation(C) :-
    claim.service_type(C, ivf_for_fertility_preservation),
    eligible_for_fertility_preservation(C),
    not(excluded_location(claim.location_service_received(C))).

covered(C) :-
    claim.service_type(C, cryopreservation_for_fertility_preservation),
    eligible_for_fertility_preservation(C),
    not(excluded_location(claim.location_service_received(C))).
    
covered(C) :- 
    covered_ivf_for_fertility_preservation(C).
```
