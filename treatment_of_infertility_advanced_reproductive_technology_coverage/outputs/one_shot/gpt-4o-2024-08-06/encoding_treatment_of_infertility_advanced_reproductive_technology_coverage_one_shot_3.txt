    claim.patient_is_believed_to_be_fertile(C, yes),

ovarian_success_likelihood(C) :- 
    claim.patient_has_premature_ovarian_failure(C, no),
    claim.age(C, Age),
    claim.fsh_level(C, FSHLevel),
    (
        (Age < 40, FSHLevel < 19);
        (Age >= 40, FSHLevel < 19)
    ).
