covered(C) :-  
    % Must be one of the listed ART services for fertility preservation
    ( claim.service_type(C, ivf_for_fertility_preservation)
    ; claim.service_type(C, cryopreservation_for_fertility_preservation) ),

    % Must be received in-network (ART services not covered out-of-network)
    ( claim.location_service_received(C, stanford_health_care)
    ; claim.location_service_received(C, menlo_medical_clinic)
    ; claim.location_service_received(C, sutter_health_palo_alto_medical_foundation)
    ; claim.location_service_received(C, tier_two_in_network) ),

    % Must meet fertility preservation eligibility
    claim.patient_is_believed_to_be_fertile(C, yes),
    ( claim.planned_service(C, chemotherapy)
    ; claim.planned_service(C, pelvic_radiotherapy)
    ; claim.planned_service(C, gonadotoxic_therapy)
    ; claim.planned_service(C, orchiectomy)
    ; claim.planned_service(C, oophorectomy)
    ; claim.planned_service(C, hysterectomy)
    ; claim.planned_service(C, orchiectomy_for_treatment_of_disease)
    ; claim.planned_service(C, oophorectomy_for_treatment_of_disease)
    ; claim.planned_service(C, other_treatment_demonstrated_to_result_in_infertility) ),

    % Must satisfy ovarian reserve criteria if retrieving eggs
    ( claim.will_be_retrieving_eggs_for_use(C, no)
    ; ( claim.will_be_retrieving_eggs_for_use(C, yes),
        % Incorporate Aetna Infertility Clinical Policy & Student Health Plan criteria:
        % Age-based FSH/ovarian responsiveness checks, or premature ovarian failure rules, etc.
        % Here assumed as a single atom for meeting those external criteria:
        fact(meets_aetna_infertility_ovarian_reserve_criteria(C))
      )
    ),

    % Check any additional external policy requirements (e.g., age, trial criteria, POF up to age 45)
    fact(meets_aetna_infertility_clinical_policy_criteria(C)),

    % Must not exceed planâ€™s cycle/course limit
    claim.num_other_covered_art_services(C, UsedCycles),
    UsedCycles < 1.
