```prolog
covered(C) :-
    % Policy must be in effect
    claim_agreement_signed(C, yes),
    claim_premium_paid(C, yes),
    claim_hospitalization_occurred_months_from_effective_date(C, HMonths),
    (
        claim_wellness_visit_occurred_months_from_effective_date(C, WMonths),
        claim_wellness_visit_confirmation_provided_months_from_effective_date(C, WCMonths),
        HMonths =< 6;
        (HMonths > 6, WMonths =< 6, WCMonths =< 7)
    ),
    claim_fraud_occurred(C, no),
    claim_misrepresentation_occurred(C, no),
    claim_material_witholding_occurred(C, no),
    claim_hospitalization_occurred_months_from_effective_date(C, HMonths), HMonths =< 12,

    % Hospitalization reasons must align with policy's terms
    (claim_hospitalization_reason(C, sickness); claim_hospitalization_reason(C, accidental_injury)),

    % Exclusions
    claim_hospitalization_causal_event(C, CausalEvent),
    CausalEvent \= skydiving,
    CausalEvent \= military_service,
    CausalEvent \= firefighting_service,
    CausalEvent \= police_service,
    claim_patient_age(C, Age),
    Age < 80.
```
