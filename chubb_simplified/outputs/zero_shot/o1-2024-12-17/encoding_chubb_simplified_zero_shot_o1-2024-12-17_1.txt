covered(C) :-
    claim_hospitalization_reason(C, R),
    (R = sickness; R = accidental_injury),
    claim_hospitalization_occurred_months_from_effective_date(C, M),
    M =< 12,
    claim_agreement_signed(C, yes),
    claim_premium_paid(C, yes),
    \+ claim_fraud_occurred(C, yes),
    \+ claim_misrepresentation_occurred(C, yes),
    \+ claim_material_witholding_occurred(C, yes),
    claim_wellness_visit_occurred_months_from_effective_date(C, WVisit),
    claim_wellness_visit_confirmation_provided_months_from_effective_date(C, WConfirm),
    claim_hospitalization_causal_event(C, Event),
    Event = other,
    claim_country_of_sickness_or_injury(C, Country),
    (Country = usa; Country = non_usa),
    claim_patient_age(C, Age),
    Age < 80,
    (
        M =< 7
        ;
        (
            WVisit =< 6,
            WConfirm =< 7
        )
    ).
