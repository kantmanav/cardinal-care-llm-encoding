covered(C) :- 
    claim_service_type(C, ServiceType),
    claim_location_service_received(C, Location),
    claim_patient_is_believed_to_be_fertile(C, FertilityBelief),
    claim_planned_service(C, PlannedService),
    claim_will_be_retrieving_eggs_for_use(C, RetrievingEggs),
    patient_age_at_time_of_service(C, Age),
    claim_patient_has_premature_ovarian_failure(C, PrematureOvarianFailure),
    
    % Check ART services criteria
    ((ServiceType = ivf_for_fertility_preservation ; 
      ServiceType = cryopreservation_for_fertility_preservation),
     
     % Check eligibility for fertility preservation
     FertilityBelief = yes,
     (PlannedService = chemotherapy ;
      PlannedService = pelvic_radiotherapy ;
      PlannedService = gonadotoxic_therapy ;
      PlannedService = orchiectomy ;
      PlannedService = oophorectomy ;
      PlannedService = hysterectomy ;
      PlannedService = orchiectomy_for_treatment_of_disease ;
      PlannedService = oophorectomy_for_treatment_of_disease ;
      PlannedService = other_treatment_demonstrated_to_result_in_infertility),
     
     % Check ART services location
     (Location = stanford_health_care ;
      Location = menlo_medical_clinic ;
      Location = sutter_health_palo_alto_medical_foundation ;
      Location = tier_two_in_network),
     
     % Ovarian reserve and IVF success likelihood check only if retrieving eggs
     RetrievingEggs = no ;
     (RetrievingEggs = yes,
      most_recent_day_3_fsh_test(C, fsh_test(Level, Date, _)),
      (Age < 40, Level < 19 ;
       Age >= 40, 
       day_3_fsh_tests_since_age_40_in_claim(C, Tests), 
       \+ exists_test_with_level_at_least_19(Tests),
       leq_n_months_diff(Date, Date, 6) ;
       PrematureOvarianFailure = yes, Level < 19, 
       leq_n_months_diff(Date, Date, 6))))
