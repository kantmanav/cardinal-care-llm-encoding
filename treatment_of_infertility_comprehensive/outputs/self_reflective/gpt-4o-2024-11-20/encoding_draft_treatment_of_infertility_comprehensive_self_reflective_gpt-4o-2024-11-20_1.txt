covered(C) :-
    % Check the service type is covered
    (   claim_service_type(C, ovulation_induction_cycle_with_menotropins)
    ;   claim_service_type(C, intrauterine_insemination)
    ),
    % Patient must be infertile, verified by patient_is_infertile/1
    patient_is_infertile(C),
    
    % Patient or partner must not have had voluntary sterilization
    claim_patient_has_had_voluntary_sterilization(C, no),
    
    % Condition causing infertility must not be due to natural physiologic processes
    \+ claim_demonstrated_and_documented_condition_causing_infertility(C, natural_menopause),
    \+ claim_demonstrated_and_documented_condition_causing_infertility(C, natural_perimenopause),
    \+ claim_demonstrated_and_documented_condition_causing_infertility(C, natural_male_reproductive_aging),
    \+ claim_demonstrated_and_documented_condition_causing_infertility(C, other_natural_physiologic_process),

    % A successful pregnancy cannot be attained through less costly treatments
    claim_successful_pregnancy_can_be_attained_through_less_costly_treatment_covered_by_plan(C, no),

    % Check additional criteria based on the patient and their partner
    claim_sex_of_patient_trying_to_conceive(C, PatientSex),
    claim_sex_of_patients_partner(C, PartnerSex),
    (   PatientSex = female ->
        patient_age_at_time_of_service(C, Age),
        most_recent_day_3_fsh_test(C, fsh_test(FSHLevel, FSHDate, _Time)),
        (   Age < 35 ->
            (   PartnerSex = male ->
                % Female under 35 with male partner
                (   claim_num_months_timed_unprotected_intercourse(C, Months),
                    Months >= 12
                ;   claim_num_cycles_donor_insemination(C, Cycles),
                    Cycles >= 12
                ),
                leq_n_months_diff(FSHDate, claim_date_service_received(C), 12),
                FSHLevel < 19
            ;   PartnerSex = no_partner ->
                % Female under 35 without male partner
                claim_num_cycles_donor_insemination(C, Cycles),
                Cycles >= 12,
                leq_n_months_diff(FSHDate, claim_date_service_received(C), 12),
                FSHLevel < 19
            )
        ;   Age >= 35 ->
            (   PartnerSex = male ->
                % Female 35 or older with male partner
                (   claim_num_months_timed_unprotected_intercourse(C, Months),
                    Months >= 6
                ;   claim_num_cycles_donor_insemination(C, Cycles),
                    Cycles >= 6
                ),
                (   Age < 40 ->
                    leq_n_months_diff(FSHDate, claim_date_service_received(C), 6),
                    FSHLevel < 19
                ;   Age >= 40 ->
                    day_3_fsh_tests_since_age_40_in_claim(C, Tests),
                    \+ exists_test_with_level_at_least_19(Tests)
                )
            ;   PartnerSex = no_partner ->
                % Female 35 or older without male partner
                claim_num_cycles_donor_insemination(C, Cycles),
                Cycles >= 6,
                (   Age < 40 ->
                    leq_n_months_diff(FSHDate, claim_date_service_received(C), 6),
                    FSHLevel < 19
                ;   Age >= 40 ->
                    day_3_fsh_tests_since_age_40_in_claim(C, Tests),
                    \+ exists_test_with_level_at_least_19(Tests)
                )
            )
        )
    ;   PatientSex = male ->
        (   claim_age_of_patients_female_partner(C, PartnerAge),
            (   PartnerAge < 35 ->
                claim_num_months_timed_unprotected_intercourse(C, Months),
                Months >= 12
            ;   PartnerAge >= 35 ->
                claim_num_months_timed_unprotected_intercourse(C, Months),
                Months >= 6
            )
        )
    ).
