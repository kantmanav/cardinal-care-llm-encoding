covered(C) :-
    claim.service_type(C, female_contraceptive_counseling),
    claim.type_of_contraceptive_counseling(C, TC),
    (TC = methods ; TC = management_of_side_effects ; TC = adherence),
    claim.counseling_setting(C, CS),
    (CS = individual ; CS = group).

covered(C) :-
    claim.service_type(C, female_contraceptive_drugs_or_devices_and_follow_up_care),
    claim.type_of_female_contraceptive_method_provided(C, M),
    fda_approved_female_contraceptive_method(M),
    ( \+ otc_female_contraceptive_method(M)
    ;
      (otc_female_contraceptive_method(M), claim.otc_female_contraceptive_method_is_prescribed_by_provider(C, yes))
    ).

covered(C) :-
    claim.service_type(C, female_voluntary_sterilization),
    claim.type_of_female_voluntary_sterilization_procedure_service_or_supplies(C, T),
    ( T = tubal_ligation ; T = sterilization_implants ; T = related_services ; T = related_supplies ).
