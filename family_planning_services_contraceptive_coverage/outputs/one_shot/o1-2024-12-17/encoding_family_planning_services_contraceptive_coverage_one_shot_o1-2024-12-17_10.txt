covered(C) :-
    (   % Counseling services
        claim.service_type(C, female_contraceptive_counseling),
        claim.type_of_contraceptive_counseling(C, X),
        (X = methods
        ; X = management_of_side_effects
        ; X = adherence),
        claim.counseling_setting(C, S),
        (S = individual
        ; S = group)
    )
    ;
    (   % Female contraceptive drugs/devices and follow-up care
        claim.service_type(C, female_contraceptive_drugs_or_devices_and_follow_up_care),
        claim.type_of_female_contraceptive_method_provided(C, M),
        (   fda_approved_female_contraceptive_method(M)
        ;   (   otc_female_contraceptive_method(M),
                claim.otc_female_contraceptive_method_is_prescribed_by_provider(C, yes)
            )
        ),
        claim.mode_of_providing_service(C, T),
        (T = provided_by
        ; T = administered_by
        ; T = removed_by
        ; T = provided_related_service
        ; T = provided_related_supplies
        ; T = provided_follow_up_care)
    )
    ;
    (   % Female voluntary sterilization
        claim.service_type(C, female_voluntary_sterilization),
        claim.type_of_female_voluntary_sterilization_procedure_service_or_supplies(C, P),
        (P = tubal_ligation
        ; P = sterilization_implants
        ; P = related_services
        ; P = related_supplies)
    ).
