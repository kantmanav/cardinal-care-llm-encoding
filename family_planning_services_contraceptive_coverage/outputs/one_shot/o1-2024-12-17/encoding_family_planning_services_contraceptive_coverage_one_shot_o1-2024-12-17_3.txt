covered(C) :-
  % Counseling services
  claim.service_type(C, female_contraceptive_counseling),
  (
    claim.type_of_contraceptive_counseling(C, methods);
    claim.type_of_contraceptive_counseling(C, management_of_side_effects);
    claim.type_of_contraceptive_counseling(C, adherence)
  ),
  (
    claim.counseling_setting(C, group);
    claim.counseling_setting(C, individual)
  ).

covered(C) :-
  % Contraceptive drugs/devices and follow-up care
  claim.service_type(C, female_contraceptive_drugs_or_devices_and_follow_up_care),
  (
    (
      (
        claim.mode_of_providing_service(C, provided_by);
        claim.mode_of_providing_service(C, administered_by);
        claim.mode_of_providing_service(C, removed_by);
        claim.mode_of_providing_service(C, provided_related_service);
        claim.mode_of_providing_service(C, provided_related_supplies)
      ),
      claim.type_of_female_contraceptive_method_provided(C, M),
      fda_approved_female_contraceptive_method(M),
      (
        not(otc_female_contraceptive_method(M));
        claim.otc_female_contraceptive_method_is_prescribed_by_provider(C, yes)
      )
    );
    claim.mode_of_providing_service(C, provided_follow_up_care)
  ).

covered(C) :-
  % Female voluntary sterilization
  claim.service_type(C, female_voluntary_sterilization),
  claim.type_of_female_voluntary_sterilization_procedure_service_or_supplies(C, S),
  (
    S = tubal_ligation;
    S = sterilization_implants;
    S = related_services;
    S = related_supplies
  ).
